10.A program that reads the student's name and marks in five subjects. Then calculate and print: Total marks (sum of all five subjects), Average marks (sum divided by 5, rounded to two decimal places)

public class Main {
    public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    String name =sc.next();
    int m1 = sc.nextInt();
    int m2 = sc.nextInt();
    int m3 = sc.nextInt();
    int m4 = sc.nextInt();
    int m5 = sc.nextInt();

    int tot = m1+m2+m3+m4+m5;
    double avg= tot/5.0;

    System.out.printf("Average Marks: %.2f\n",avg);
  }
}
