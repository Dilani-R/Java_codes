1.Take an integer as input and return "POSITIVE" if the integer is positive, "NEGATIVE" if the integer is negative and "ZERO" if the integer is zero

import java.util.Scanner;
public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int num = sc.nextInt();
        if (num > 0) {
            System.out.println("POSITIVE");
        } else if (num < 0) {
            System.out.println("NEGATIVE");
        } else {
            System.out.println("ZERO");
        }
    }
}

            System.out.println("No second max");

            System.out.println(a);

        } else {

6.Triangle Angle Validation

import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int A = sc.nextInt();
        int B = sc.nextInt();
        int C = sc.nextInt();

        if (A + B + C == 180 && A > 0 && B > 0 && C > 0) {
            System.out.println("VALID");

            if (A == 60 && B == 60 && C == 60)
                System.out.println("EQUILATERAL");
            else if (A == B || B == C || A == C)
                System.out.println("ISOSCELES");
            else
                System.out.println("SCALENE");
        } else {
            System.out.println("NOT VALID");
        }
    }
}

9.Fun Grade Calculator

import java.util.*;

        int total = 0;
        for (int i = 0; i < 5; i++) {
            int mark = sc.nextInt();
            total += mark;
        }

        double average = total / 5.0;

        if (average >= 90) {
            System.out.println("Excellent!");
        } else if (average >= 80) {
            System.out.println("Great Job!");
        } else if (average >= 70) {
            System.out.println("Good Effort!");
        } else if (average >= 50) {
            System.out.println("Keep Trying!");
        } else {
            System.out.println("Better Luck Next Time!");
        }
    }
}

11.Electricity Bill Calculator

import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        double units = sc.nextDouble();
        double bill = 0;

        if (units <= 50) {
            bill = units * 1;
        } else if (units <= 150) {
            bill = 50 * 1 + (units - 50) * 2;
        } else if (units <= 250) {
            bill = 50 * 1 + 100 * 2 + (units - 150) * 3;
        } else {
            bill = 50 * 1 + 100 * 2 + 100 * 3 + (units - 250) * 4;
        }

        // Add 20% surcharge if total bill > 150
        if (bill > 150) {
            bill += bill * 0.2;
        }

        // Print as integer if no decimal part
        System.out.println((int)bill);
    }
}

13.Train Fare Calculator

import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int age = sc.nextInt();
        int distance = sc.nextInt();
        double standardFare = 5;
        double fare = distance * standardFare;

        if (age < 12) {
            fare *= 0.5; // 50% fare for children
        } else if (age >= 60) {
            fare *= 0.75; // 75% fare for seniors
        } // Adults pay 100%, so no change

        // Print fare (as integer if whole number)
        if (fare == (int)fare) {
            System.out.println((int)fare);
        } else {
            System.out.println(fare);
        }
    }
}

15.Fun Character Case Converter

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        char ch = sc.next().charAt(0);

        if (ch >= 'a' && ch <= 'z') {
            System.out.println((char)(ch - 'a' + 'A'));
        } else if (ch >= 'A' && ch <= 'Z') {
            System.out.println((char)(ch - 'A' + 'a'));
        } else {
            System.out.println("abcd theriyatha?");
        }
    }
}
